    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        
         if(FirstCountryName.getSelectedIndex()==0 || SecondCountryName.getSelectedIndex()==0 || jTextField1.getText().equals("")){
             JOptionPane.showMessageDialog(this,"Please Add The Input","ERROR MESSAGE", JOptionPane.INFORMATION_MESSAGE);
             return;
         }
        
         double amount=Double.parseDouble(jTextField1.getText());
         double amountToPound = 0.00;
         
        
         
         switch(FirstCountryName.getSelectedItem().toString()){
             case "USA" : amountToPound = amount / us_dollar;
             break;
             case "Nigeria" :amountToPound = amount / nigerian_naira;
             break;
             case "Brazil" :amountToPound = amount / brazilian_real;
             break;
             case "Canada" :amountToPound = amount / canadian_dollar;
             break;
             case "Indonesia" :amountToPound = amount / indonesia_rupiah;
             break;
             case "India" :amountToPound = amount / indian_rupee;
             break;
             case "Pakistan" :amountToPound = amount / pakistani_rupee;
             break;
             
             
         }
         double amountchange =0.0;
         switch(SecondCountryName.getSelectedItem().toString()){
             case "USA" : amountchange = amountToPound * us_dollar;
             break;
             case "Nigeria" : amountchange = amountToPound * nigerian_naira;
             break;
             case "Brazil" : amountchange = amountToPound * brazilian_real;
             break;
             case "Canada" : amountchange = amountToPound * canadian_dollar;
             break;
             case "Indonesia" : amountchange = amountToPound * indonesia_rupiah;
             break;
             case "India" : amountchange = amountToPound * indian_rupee;
             break;
             case "Pakistan" : amountchange = amountToPound * pakistani_rupee;
             break;
             
            
         }
             String value = String.format("%.2f", amountchange);
             
             jTextField2.setText(value);
    }
